-- equi join
-- 가장 흔한 조인으로 where Equal(=) 조건으로 테이블을 조인함
select a.nickname, ar.review from firstschema.audience a, firstschema.audience_review ar
where a.id = ar.audience_id;

-- non equi join
-- where Equal(=) 외의 다른 조건으로 조인하는 방식
select a.nickname, s.performance_name, to_char(s.performance_date, 'YYYY-MM-DD') from firstschema.audience a, firstschema.store s
where a.ticket_id = s.ticket_id -- 보통 equal이 들어감 안들어가면 데이터가 중복 출력됨
and s.performance_date between now() and date_trunc('year', now() + interval '1 year' - interval '1 day') -- end of year

-- 3개 조인
select a.nickname,
s.performance_name,
ar.review
from firstschema.audience a,
firstschema.audience_review ar,
firstschema.store s
where a.ticket_id = s.ticket_id
and a.id = ar.id

-- oracle outer join
-- 조건에 만족하지 않는 행들도 출력한다.
-- oracle에서는 (+)로 ourter 조인임을 명시할 수 있다.
select a.nickname, s.performance_name, to_char(s.performance_date, 'YYYY-MM-DD') from firstschema.audience a, firstschema.store s
where a.ticket_id = s.ticket_id -- 보통 equal이 들어감 안들어가면 데이터가 중복 출력됨
and s.performance_date between now() and date_trunc('year', now() + interval '1 year' - interval '1 day') -- end of year

-- standard join
-- 벤더 표준 ansi sql 조인

--- inner join
--- euqal join과 흡사하나 테이블을 합친다는 점을 명시적으로 표기하기에 inner join이 더 선호됨
select a.nickname, ar.review from firstschema.audience a
inner join firstschema.audience_review ar
on a.id = ar.audience_id;

--- outer join
--- left outer join : 왼쪽의 데이터는 무조건 출력됨
--- right ourter join : 오른쪽의 데이터는 무조건 출력됨
select count(\*)::text || '명' -- 90명
from firstschema.audience a
left outer join firstschema.audience_review ar
on a.id = ar.audience_id;

select count(\*)::text || '명' -- 30명
from firstschema.audience_review ar
left outer join firstschema.audience a
on a.id = ar.audience_id;

--- right ourter join: 오른쪽의 데이터는 무조건 출력됨
select count(\*)::text || '명' -- 30명
from firstschema.audience a
right outer join firstschema.audience_review ar
on a.id = ar.audience_id;

select count(\*)::text || '명' -- 90명
from firstschema.audience_review ar
right outer join firstschema.audience a
on a.id = ar.audience_id;

--- full outer join: 합칩합이다. 단 중복값이 제거된다.
select \*
from firstschema.audience_review ar
full outer join firstschema.audience a
on ar.audience_id = a.id

--- natural join: 동일한 이름을 가진 컬럼들이 모두 동일한 데이터를 가지고 있을 경우 join되는 방식이다. mssql은 미지원
